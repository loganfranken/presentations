The Media Query

@media (expression)
{
   /* CSS */
}
@media (max-width: 600px)
{
   /* If viewport is 600px or less */
   .info
   {
      width: 50%;
   }   
}
@media (min-width: 300px)
{
   /* If viewport is 300px or more */
   .callout
   {
      float: left;
   }   
}
@media (max-device-width: 400px)
{
   /* If width of actual screen
      is 400px or less          */
   .input
   {
      width: 100%;
   }   
}
